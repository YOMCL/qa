  <action>
Create `.github/workflows/app-maestro.yml` with the following EXACT structure:

```yaml
name: Maestro Nightly — Prinorte APP

on:
  schedule:
    - cron: '0 6 * * *'  # 06:00 UTC = 03:00 CLT
  workflow_dispatch:

jobs:
  maestro:
    runs-on: ubuntu-latest
    timeout-minutes: 45
    # Only run if APK URL is configured
    if: ${{ github.event_name == 'workflow_dispatch' || github.event.schedule }}

    steps:
      - uses: actions/checkout@v4

      - name: Enable KVM
        run: |
          echo 'KERNEL=="kvm", GROUP="kvm", MODE="0666", OPTIONS+="static_node=kvm"' | sudo tee /etc/udev/rules.d/99-kvm4all.rules
          sudo udevadm control --reload-rules
          sudo udevadm trigger --name-match=kvm

      - uses: actions/setup-java@v4
        with:
          distribution: temurin
          java-version: '17'

      - name: Install Maestro CLI 1.40.0
        run: |
          export MAESTRO_VERSION=1.40.0
          curl -fsSL "https://get.maestro.mobile.dev" | bash
          echo "$HOME/.maestro/bin" >> $GITHUB_PATH

      - name: Download APK
        run: |
          if [ -z "${{ secrets.APK_URL }}" ]; then
            echo "::error::APK_URL secret not configured. Set it in repo Settings > Secrets."
            exit 1
          fi
          curl -fsSL "${{ secrets.APK_URL }}" -o app.apk
          echo "APK downloaded: $(ls -lh app.apk | awk '{print $5}')"

      - name: Run Maestro on Emulator
        uses: reactivecircus/android-emulator-runner@v2
        with:
          api-level: 29
          target: default
          arch: x86_64
          profile: pixel_2
          emulator-options: -no-window -gpu swiftshader_indirect -noaudio -no-boot-anim -no-snapshot
          disable-animations: true
          script: |
            adb install -r app.apk
            DATE=$(date +%Y-%m-%d)
            mkdir -p "QA/Prinorte/${DATE}"
            maestro test \
              --env APP_PACKAGE="${{ secrets.APP_PACKAGE }}" \
              --env TEST_SELLER_EMAIL="${{ secrets.TEST_SELLER_EMAIL }}" \
              --env TEST_SELLER_PASSWORD="${{ secrets.TEST_SELLER_PASSWORD }}" \
              tests/app/flows/prinorte-session.yaml \
              2>&1 | tee "QA/Prinorte/${DATE}/maestro.log"

      - name: Generate HTML report
        if: always()
        run: |
          DATE=$(date +%Y-%m-%d)
          LOG="QA/Prinorte/${DATE}/maestro.log"
          mkdir -p public/app-reports
          if [ -f "$LOG" ]; then
            python3 tools/generate-maestro-report.py \
              "Prinorte" "prinorte" "${DATE}" \
              "${LOG}" "public/app-reports/prinorte-${DATE}.html" \
              "public/app-reports/manifest.json" "prinorte-${DATE}.html" "ci"
          else
            echo "::warning::No maestro.log found — skipping report generation"
          fi

      - name: Upload artifacts
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: maestro-report-${{ github.run_id }}
          path: |
            QA/Prinorte/*/maestro.log
            public/app-reports/prinorte-*.html
          retention-days: 30

      - name: Notify Slack on failure
        if: failure() && env.SLACK_MAESTRO_WEBHOOK != ''
        uses: slackapi/slack-github-action@v1.27.0
        with:
          payload: |
            {
              "text": "Maestro CI failed — Prinorte APP flows\nSee: ${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}"
            }
        env:
          SLACK_WEBHOOK_URL: ${{ secrets.SLACK_MAESTRO_WEBHOOK }}
          SLACK_WEBHOOK_TYPE: INCOMING_WEBHOOK

      - name: Create GitHub issue on failure (fallback)
        if: failure() && env.SLACK_MAESTRO_WEBHOOK == ''
        run: |
          gh issue create \
            --title "Maestro CI failed — Prinorte $(date +%Y-%m-%d)" \
            --body "Nightly Maestro run failed. See: ${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}" \
            --label "qa,maestro"
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          SLACK_MAESTRO_WEBHOOK: ${{ secrets.SLACK_MAESTRO_WEBHOOK }}
```

Key implementation notes:
- The `if:` condition on the Slack step checks if the webhook env var is non-empty. Use `env.SLACK_MAESTRO_WEBHOOK` (set from secrets) to gate Slack vs GitHub issue fallback.
- The `Download APK` step fails explicitly with `::error::` annotation if `APK_URL` is not configured — this prevents silent no-op runs.
- `DATE` must be computed independently in each step (GitHub Actions steps don't share shell variables).
- The `script:` block in `android-emulator-runner` runs INSIDE the emulator's shell context — `adb install` and `maestro test` execute while the emulator is running.
- Use `2>&1 | tee` to capture both stdout and stderr to the log file while showing output in Actions log.
- The Slack notification condition `failure() && env.SLACK_MAESTRO_WEBHOOK != ''` ensures Slack is only attempted if the webhook secret is set. The GitHub issue fallback uses the inverse condition.
